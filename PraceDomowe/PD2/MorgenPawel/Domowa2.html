get_argument_name <- function(original){
  # Funkcja zwraca nazwÄ™ nowego argumentu, ktÃ³ra nie kÅ‚Ã³ci siÄ™ z nazwami argumentÃ³w funkcji original.
  # DomyÅ›lnie: "argslist".
  # WystÄ™puje we wszystkich wersjach.
  argslist_name <- "argslist"
  while(argslist_name %in% names(get_formals(original))) 
    argslist_name <- paste0(argslist_name, "1")
  argslist_name
}

fix_argument_name <- function(new_name, FUNCTION){
  # Funkcja zastÄ™puje nazwÄ™ pierwszego argumentu funkcji FUNCTION (argslist) nazwÄ… new_name.
  # ZakÅ‚adamy, Å¼e w ciele funkcji jedyne uÅ¼ycia argslist odwoÅ‚ujÄ… siÄ™Â do jej pierwszego argumentu.
  # WystÄ™puje w wersji 1 i 2.
  new_body <- capture.output(body(FUNCTION)) %>%
    stri_replace_all_fixed("argslist", new_name) %>%
    str2expression()
  body(FUNCTION) <- new_body[[1]]
  names(fn_fmls(FUNCTION))[1] <- new_name
  FUNCTION
}

concat_quotes <- function(...){
  # Funkcja przyjmuje obiekty typu expression i je konkatenuje w jeden obiekt typu expression otoczony {}.
  quotes <- list(...) %>%
    lapply(function(q) capture.output(q)) 
  str2expression(c("{",do.call(c, quotes), "}"))[[1]]
}
  

get_formals <- function(FUNCTION){
  # Funkcja przyjmuje na wejÅ›ciu dowolnÄ… funkcjÄ™ i zwraca listÄ™ jej argumentÃ³w.
  # WystÄ™puje w wersji 2 i 3.
  if(!is_primitive(FUNCTION)) return(fn_fmls(FUNCTION))
  # Zaczynamy zabawÄ™.
  # Åapiemy to, co wypisuje wywoÅ‚anie FUNCTION i robimy z tego jeden ogromny string.
  fn_string <- do.call(paste, as.list(capture.output(FUNCTION))) 
    
  # Musimy znaleÅºÄ‡ segment (arg1, arg2, arg3=function1(), ...). 
  # Niestety, w Å›rodku mogÄ… siÄ™ pojawiÄ‡ wredne argumenty, ktÃ³rych wartoÅ›ciami domyÅ›lnymi 
  # sÄ… wartoÅ›ci zwracane przez wywoÅ‚anie jakiejÅ› funkcji - przez to nie moÅ¼emy po prostu 
  # poszukaÄ‡ zamykajÄ…cego nawiasu. Robienie tego regexami byÅ‚oby zbyt trudne (koniecznoÅ›Ä‡ rekurencji), 
  # wiÄ™c zastosowaÅ‚em inny sposÃ³b. 
    
  open_brackets <- stri_locate_all(fn_string, fixed = "(")[[1]][,1]  
  closed_brackets <- stri_locate_all(fn_string, fixed = ")")[[1]][,1]  
  names(open_brackets) <- rep("open", length(open_brackets))
  names(closed_brackets) <- rep("closed", length(closed_brackets))
  sorted_brackets <- c(open_brackets, closed_brackets) %>% sort()  
  assign_weights <- function(str) {
    str[str=="open"]<-1
    str[str=="closed"]<- -1
    as.numeric(str)
  }
  summed_brackets <- assign_weights(names(sorted_brackets)) %>% cumsum()  
  # Szukamy w summed_brackets 0. Indeks, na ktÃ³rym bÄ™dzie, to bÄ™dzie indeks obiektu sorted_brackets, 
  # na ktÃ³rym stoi pozycja szukanego przez nas nawiasu zamykajÄ…cego.
  summed_brackets[summed_brackets >= 10] <- 9
  # Stworzymy z summed_brackets stringa, ktÃ³ry bÄ™dzie ciÄ…giem cyfr. 
  # Nie interesujÄ… nas dokÅ‚adne wartoÅ›ci liczb powyÅ¼ej 10 - moÅ¼emy je zmapowaÄ‡ na 9. 
  # DziÄ™ki temu i-ty znak w stringu bÄ™dzie odpowiadaÅ‚ i-tej liczbie z summed_brackets.
  brackets_as_int <- do.call(paste0, as.list(summed_brackets)) %>%
    stri_extract_first_regex(pattern = "1[1-9]*0") 
  # brackets_as_int niesie informacje o nawiasach wewnÄ…trz listy argumentÃ³w funkcji.
  # inner_brackets - znajduje wszystkie pozycje startu i koÅ„ca wywoÅ‚aÅ„ funkcji wewnÄ…trz listy argumentÃ³w.
  inner_brackets <- stri_locate_all_regex(brackets_as_int,
                                          pattern = "2[2-9]*1", 
                                          omit_no_match = TRUE)[[1]]
  all_comas <- stri_locate_all_fixed(fn_string, ",", omit_no_match = TRUE)[[1]][,1] 
  # substring indexes przechowuje informacjÄ™ o pozycjach, na ktÃ³rych zaczyna siÄ™ i koÅ„czy lista argumentÃ³w.
  substring_indexes <- sorted_brackets[c(1, stri_length(brackets_as_int))] 
  all_comas <- all_comas[all_comas < substring_indexes[2]] 
  # jeÅ›li wewnÄ…trz listy argumentÃ³w nie ma wywoÅ‚aÅ„ innych funkcji, to kaÅ¼dy przecinek oddziela od siebie
  # argumenty. JeÅ›li sÄ… - pewne przecinki mogÄ… wystÄ™powaÄ‡ wewnÄ…trz wywoÅ‚aÅ„ innych funkcji:
   if(length(inner_brackets)!=0){
     forbidden_chars <- do.call(c, apply(inner_brackets, 1, 
                      function(row) sorted_brackets[row[1]]:sorted_brackets[row[2]]))  
     all_comas <- all_comas[!all_comas %in% forbidden_chars]  
   }
  # moÅ¼e siÄ™ zdarzyÄ‡, Å¼e nasza funkcja nie ma przecinkÃ³w w liÅ›cie argumentÃ³w. 
  # Oznacza to, Å¼e ma tylko jeden argument.
   if(length(all_comas) != 0){
     begins <- as.list(c(substring_indexes[1], all_comas))
     # Ostatni string (odpowiadajÄ…cy ostatniemu argumentowi na liÅ›cie) niech bÄ™dzie bez ostatniego nawiasu.
     # PÃ³Åºniej go doÅ‚Ä…czymy
     ends <- as.list(c(all_comas-1, substring_indexes[2]-1))
     parted_arguments <- stri_sub_all(fn_string, begins, ends) 
   }
   else parted_arguments <- list(stri_sub(fn_string, substring_indexes[1], substring_indexes[2]-1))
   
    # Oto powÃ³d caÅ‚ego zamieszania. Plan byÅ‚ taki, by sztucznie skonstruowaÄ‡ wywoÅ‚anie funkcji alist(), 
    # ale ona w miejscach argumentÃ³w bez wartoÅ›ci domyÅ›lnej musi przyjÄ…Ä‡ input postaci "arg1=", nie "arg1".
    # ZnajdÅºmy zatem argumenty bez wartoÅ›ci domyÅ›lnej (bez "=") i doÅ‚Ä…czmy do nich "=".
   parted_arguments[!stri_detect_fixed(parted_arguments, "=")] <- parted_arguments[!stri_detect_fixed(
      parted_arguments, "=")] %>%
     lapply(function(str) paste(str, "= "))
   # DoÅ‚Ä…czamy nawias i v'oila!
   parted_arguments <- append(parted_arguments, ")") 
   formals_string <- do.call(paste0, parted_arguments)
   paste0("alist", formals_string) %>% parse_expr() %>% eval()
}

1. Wersja pierwsza

PomysÅ‚ jest prosty - zwracana funkcja przyjmuje 2 argumenty: argslist - listÄ™, data.frame lub coÅ› Å›rodowiskopodobnego (domyÅ›lnie: Å›rodowisko wzywajÄ…cego funkcjÄ™ FUNCTION) i elipsÄ™.

OczywiÅ›cie, moÅ¼e siÄ™ zdarzyÄ‡, Å¼e funkcja FUNCTION wÅ›rÃ³d swoich argumentÃ³w ma argument o nazwie argslist. WÃ³wczas przy pomocy funkcji pomocniczych get_argument_name() i fix_argument_name() nastÄ™puje poprawka. Tym niemniej, takie sytuacje naleÅ¼Ä… do rzadkoÅ›ci - dla ustalenia uwagi w dalszej czÄ™Å›ci pracy zakÅ‚adam, Å¼e pierwszy argument rzeczywiÅ›cie nazywa siÄ™ argslist.

NSE_factory_1 <- function(FUNCTION){
  stopifnot(is_function(FUNCTION))
  out <- function(argslist = caller_env(), ...){
    given_args <- quos(...)
    evaluated_args <- lapply(given_args, function(expr) eval(quo_squash(expr), argslist))
    do.call(FUNCTION, evaluated_args)
  }
  correct_argument_name <- get_argument_name(FUNCTION) 
  if(correct_argument_name != "argslist")
    out <- fix_argument_name(correct_argument_name, out)
  out
}

Wersja druga funkcji zwraca funkcjÄ™, ktÃ³ra na pierwszym miejscu ma argument argslist - jak w wersji pierwszej. PozostaÅ‚e argumenty sÄ… identyczne, jak funkcji FUNCTION.

NSE_factory_2 <- function(FUNCTION){
  stopifnot(is_function(FUNCTION))
  arguments <- get_formals(FUNCTION)
  # DoÅ‚Ä…czamy do listy argumentÃ³w argslist z wartoÅ›ciÄ… domyÅ›lnÄ… "caller_env()".
  arguments <- append(arguments, call("caller_env"), 0)
  names(arguments)[[1]] <- "argslist"
  out <- function() {
    # Jest problem. Jako programista, nie wiem, jakie argumenty przyjmuje funkcja out.
    # Poszukamy ich w Å›rodowisku funkcji.
    arguments_names <- env_names(current_env())[-1]
    # Ignorujemy argumenty argslist i elipsÄ™.
    for(name in arguments_names[arguments_names != "..."]){
      # Sytuacja wyglÄ…da tak: name <- arg1 <- val1. JeÅ›li np ktoÅ› wywoÅ‚aÅ‚ funkcjÄ™ f(argslist, x = a), 
      # to name jest referencjÄ… na x, x jest referencjÄ… na obiekt a, 
      # ktÃ³ry trzeba zewaluowaÄ‡ w Å›rodowisku argslist.
      name <- parse_expr(name)
      new_name <- paste0("substitute(", 
             deparse(expr_interp(name)), ")") %>% parse_expr() 
      # W naszym przykÅ‚adzie new_name <- substitute(x)
      if(eval(new_name) %>% is_missing()) next
      # eval(new_name) to niezewaluowany obiekt a
      new_value <- eval(eval(new_name) ,argslist)
      if(is.name(new_value)) next
      assign(x = deparse(name),value = new_value)
      # teraz w naszym Å›rodowisku mieszka nowa zmienna poprawnie zewaluowana. Dalej
    }
    # Teraz wszystkie zmienne sÄ… dobrze zewaluowane, poza ewentualnÄ… elipsÄ…. Zbieramy je do kupy
    evaluated_arguments <- as.list(current_env())[arguments_names[arguments_names != "..."]] 
    # Ewaluujemy elipsÄ™ sposobem z wersji pierwszej
    if("..." %in% arguments_names){
      arguments_extra <- quos(...)
      arguments_extra_evaluated <- lapply(arguments_extra, 
             function(expr) eval(quo_squash(expr), argslist))
      # doÅ‚Ä…czamy do listy argumentÃ³w argumenty z elipsy, na dobre miejsce
      evaluated_arguments <- append(evaluated_arguments,
             arguments_extra_evaluated,
             which(arguments_names == "...")) 
    }
    do.call(FUNCTION, evaluated_arguments)
  }
  formals(out) <- arguments
  correct_argument_name <- get_argument_name(FUNCTION) 
  if(correct_argument_name != "argslist") out <- fix_argument_name(correct_argument_name, out)
  out
}
